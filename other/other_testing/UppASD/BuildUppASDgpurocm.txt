# Build instructions for UppASD on Dardel AMD GPU nodes

# Load the PrgEnv-gnu environment, ROCM, and FFTW
ml PrgEnv-gnu
ml rocm/6.3.3
ml cray-fftw/3.3.10.9
ml cmake/4.1.2
ml craype-accel-amd-gfx90a

# Create build directory
mkdir buildGnuHIP1
cd buildGnuHIP1

# Configure
cmake .. -DUSE_HIP=1

# Build
make -j 32
